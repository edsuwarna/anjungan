Disk Cleanup — Docker & Logs

Remove unused Docker images, prune logs, and clean temp files

#!/bin/bash # Disk Cleanup Script # Usage: ./cleanup.sh [threshold_pct] THRESHOLD=${1:-80} echo "=== Disk Usage Before ===" df -h / DISK_USAGE=$(df / | awk 'NR==2 {print $5}' | sed 's/%//') if [ "$DISK_USAGE" -gt "$THRESHOLD" ]; then docker system prune -af --volumes journalctl --vacuum-time=3d echo "✅ Cleanup completed" else echo "⏭️ Disk usage normal, skipping" fi
